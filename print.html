function generateReceiptImageBytes(table, discount, note, method, items, total, finalTotal, timeStr) { const canvas = document.getElementById('printCanvas'); const ctx = canvas.getContext('2d'); // 动态计算画布高度（行高调大到 32 像素以适应大字，新增单价行） let rowHeight = 32; let padding = 10; let calculatedHeight = padding * 2 + 420 + (items.length * rowHeight); if (note) calculatedHeight += 40; canvas.height = calculatedHeight; ctx.fillStyle = '#ffffff'; ctx.fillRect(0, 0, canvas.width, canvas.height); ctx.fillStyle = '#000000'; ctx.textBaseline = 'top'; let y = padding; // --- 1. 绘制 LOGO 图片 (预留位置) --- // 注意：如果需要打印图片 Logo，需要先用 JavaScript 的 Image 对象预加载你的 Firebase 图片链接 // 这里先使用优雅的大字加粗排版替代名称。如果要换成纯图片，请参考下方的图片上传教程。 ctx.font = 'bold 36px "Segoe UI", system-ui, sans-serif'; ctx.textAlign = 'center'; ctx.fillText("我的店铺Logo", 192, y); // 替换为你想要的文字或预留给图片 y += 50; // --- 2. 更改电话与地址 --- ctx.font = '20px "Segoe UI", system-ui, sans-serif'; ctx.fillText("电话: 09795858987", 192, y); y += 28; ctx.fillText("星光美食街8号", 192, y); y += 32; ctx.textAlign = 'left'; ctx.font = '18px "Segoe UI", system-ui, sans-serif'; ctx.fillText("-------------------------------------------", 4, y); y += 22; // --- 3. 优化桌号与基本信息排版 --- ctx.font = 'bold 24px "Segoe UI", system-ui, sans-serif'; ctx.fillText(`桌号：${table} 号桌`, 6, y); y += 32; ctx.font = '18px "Segoe UI", system-ui, sans-serif'; ctx.fillText(`时间：${timeStr}`, 6, y); y += 26; ctx.fillText(`支付：${method}`, 6, y); y += 26; ctx.fillText("-------------------------------------------", 4, y); y += 26; // --- 4. 重新排版表头（加入单价展示） --- ctx.font = 'bold 18px "Segoe UI", system-ui, sans-serif'; ctx.textAlign = 'left'; ctx.fillText("菜品(单价)", 6, y); ctx.fillText("数量", 210, y); ctx.textAlign = 'right'; ctx.fillText("金额(Ks)", 378, y); y += 32; // --- 5. 菜品列表大字化 + 显示单价 --- ctx.font = 'bold 18px "Segoe UI", system-ui, sans-serif'; for (let it of items) { // 在菜名后面加上 (单价) let nameWithPrice = `${it.name}(${formatNum(it.price)})`; // 如果太长则截断，防止重叠 let displayName = nameWithPrice.length > 12 ? nameWithPrice.slice(0, 11) + ".." : nameWithPrice; ctx.textAlign = 'left'; ctx.fillText(displayName, 6, y); ctx.fillText(it.qty.toString(), 220, y); ctx.textAlign = 'right'; ctx.fillText(formatNum(it.subtotal), 378, y); y += rowHeight; } ctx.textAlign = 'left'; ctx.fillText("-------------------------------------------", 4, y); y += 22; // --- 6. 账单汇总大字化 --- ctx.font = '20px "Segoe UI", system-ui, sans-serif'; ctx.fillText(`原价合计：${formatNum(total)} Ks`, 6, y); y += 28; if (discount > 0) { ctx.fillText(`优惠金额：- ${formatNum(discount)} Ks`, 6, y); y += 28; } // 显著放大实收金额 ctx.font = 'bold 26px "Segoe UI", system-ui, sans-serif'; ctx.fillText(`实收金额：${formatNum(finalTotal)} Ks`, 6, y); y += 36; ctx.font = '20px "Segoe UI", system-ui, sans-serif'; if (note) { ctx.fillText(`备注：${note}`, 6, y); y += 30; } ctx.fillText("-------------------------------------------", 4, y); y += 24; // 底部感谢语 ctx.textAlign = 'center'; ctx.font = 'bold 20px "Segoe UI", system-ui, sans-serif'; ctx.fillText("感谢光临，欢迎下次惠顾！", 192, y); // --- 以下保持原有的点阵转换逻辑不变 --- const imgData = ctx.getImageData(0, 0, canvas.width, canvas.height); const widthBytes = canvas.width / 8; const escPosBytes = []; escPosBytes.push(0x1D, 0x76, 0x30, 0x00); escPosBytes.push(widthBytes & 0xFF, (widthBytes >> 8) & 0xFF); escPosBytes.push(canvas.height & 0xFF, (canvas.height >> 8) & 0xFF); let pixelIndex = 0; for (let h = 0; h < canvas.height; h++) { for (let w = 0; w < widthBytes; w++) { let byteVal = 0; for (let b = 0; b < 8; b++) { let r = imgData.data[pixelIndex]; let g = imgData.data[pixelIndex + 1]; let bl = imgData.data[pixelIndex + 2]; let brightness = (r + g + bl) / 3; if (brightness < 128) { byteVal |= (1 << (7 - b)); } pixelIndex += 4; } escPosBytes.push(byteVal); } } return new Uint8Array(escPosBytes); }
